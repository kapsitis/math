Lemma training1_1: forall a:Prop, ~~a -> a.
Proof.
  intros a.
  unfold not.
  intros.
  elim (classic a).
  intros aTrue.
  exact aTrue.
  intros aFalse.
  pose (H aFalse) as CONTRA.
  contradiction CONTRA.
Qed.
